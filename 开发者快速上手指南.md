# 开发者快速上手指南

## 5分钟快速开始

### 1. 基本使用

```kotlin
import top.javaxiaowu.saleseasypackagingassistant.services.ServiceFactory
import top.javaxiaowu.saleseasypackagingassistant.config.ApiConfig

// 第一步: 配置API域名（可选，默认已配置）
ApiConfig.setBaseUrl("http://114.66.56.200:6002")

// 第二步: 选择服务模式
ServiceFactory.setServiceMode(useMock = false)  // false=真实API, true=模拟数据

// 第三步: 使用登录服务
val loginService = ServiceFactory.getLoginService()
val loginResult = loginService.login("username", "password")

if (loginResult.success) {
    println("登录成功!")
    println("用户: ${loginResult.userInfo?.username}")
    println("Token: ${loginResult.token}")
    
    // 第四步: 使用通知服务
    val notificationService = ServiceFactory.getNotificationService()
    
    // 获取通知列表
    val notifications = notificationService.getNotifications(page = 1, pageSize = 20)
    println("收到 ${notifications.size} 条通知")
    
    // 获取未读数量
    val unreadCount = notificationService.getUnreadCount()
    println("未读通知: $unreadCount 条")
} else {
    println("登录失败: ${loginResult.message}")
}
```

### 2. 在UI中使用

```kotlin
import com.intellij.openapi.application.ApplicationManager
import javax.swing.JButton
import javax.swing.JOptionPane

class LoginButton : JButton("登录") {
    init {
        addActionListener {
            // 在后台线程执行网络请求
            ApplicationManager.getApplication().executeOnPooledThread {
                val loginService = ServiceFactory.getLoginService()
                val result = loginService.login(usernameField.text, passwordField.text)
                
                // 在UI线程更新界面
                ApplicationManager.getApplication().invokeLater {
                    if (result.success) {
                        JOptionPane.showMessageDialog(
                            this,
                            "登录成功: ${result.userInfo?.username}",
                            "成功",
                            JOptionPane.INFORMATION_MESSAGE
                        )
                    } else {
                        JOptionPane.showMessageDialog(
                            this,
                            "登录失败: ${result.message}",
                            "错误",
                            JOptionPane.ERROR_MESSAGE
                        )
                    }
                }
            }
        }
    }
}
```

## 最佳实践

### 1. 总是在后台线程执行网络请求

```kotlin
// ✅ 正确
ApplicationManager.getApplication().executeOnPooledThread {
    val result = loginService.login(username, password)
    // 处理结果
}

// ❌ 错误 - 不要在UI线程执行网络请求
val result = loginService.login(username, password)  // 会阻塞UI
```

### 2. 检查登录状态

```kotlin
// ✅ 正确 - 先检查登录状态
fun loadUserData() {
    val loginService = ServiceFactory.getLoginService()
    if (!loginService.isLoggedIn()) {
        showLoginDialog()
        return
    }
    
    // 已登录，继续操作
    val notifications = notificationService.getNotifications()
}
```

### 3. 统一错误处理

```kotlin
// ✅ 正确 - 统一的错误处理函数
fun handleApiError(message: String?) {
    when {
        message?.contains("401") == true -> {
            // Token过期，重新登录
            showLoginDialog()
        }
        message?.contains("网络") == true -> {
            showError("网络连接失败，请检查网络")
        }
        else -> {
            showError(message ?: "未知错误")
        }
    }
}
```

### 4. 使用单例服务

```kotlin
// ✅ 正确 - 使用工厂获取服务

// ❌ 错误 - 不要直接创建服务实例

```

## 完整示例

### 登录对话框完整实现

```kotlin
import com.intellij.openapi.application.ApplicationManager
import com.intellij.openapi.ui.DialogWrapper
import javax.swing.*
import java.awt.GridBagLayout
import java.awt.GridBagConstraints

class LoginDialog : DialogWrapper(true) {
    private val usernameField = JTextField(20)
    private val passwordField = JPasswordField(20)
    
    init {
        title = "登录"
        init()
    }
    
    override fun createCenterPanel(): JComponent {
        val panel = JPanel(GridBagLayout())
        val gbc = GridBagConstraints()
        
        gbc.gridx = 0
        gbc.gridy = 0
        panel.add(JLabel("用户名:"), gbc)
        
        gbc.gridx = 1
        panel.add(usernameField, gbc)
        
        gbc.gridx = 0
        gbc.gridy = 1
        panel.add(JLabel("密码:"), gbc)
        
        gbc.gridx = 1
        panel.add(passwordField, gbc)
        
        return panel
    }
    
    override fun doOKAction() {
        val username = usernameField.text
        val password = String(passwordField.password)
        
        if (username.isEmpty() || password.isEmpty()) {
            JOptionPane.showMessageDialog(
                contentPane,
                "请输入用户名和密码",
                "错误",
                JOptionPane.ERROR_MESSAGE
            )
            return
        }
        
        // 在后台线程执行登录
        ApplicationManager.getApplication().executeOnPooledThread {
            val loginService = ServiceFactory.getLoginService()
            val result = loginService.login(username, password)
            
            // 在UI线程更新界面
            ApplicationManager.getApplication().invokeLater {
                if (result.success) {
                    super.doOKAction()  // 关闭对话框
                    JOptionPane.showMessageDialog(
                        null,
                        "登录成功: ${result.userInfo?.username}",
                        "成功",
                        JOptionPane.INFORMATION_MESSAGE
                    )
                } else {
                    JOptionPane.showMessageDialog(
                        contentPane,
                        "登录失败: ${result.message}",
                        "错误",
                        JOptionPane.ERROR_MESSAGE
                    )
                }
            }
        }
    }
}

// 使用
fun showLoginDialog() {
    val dialog = LoginDialog()
    dialog.show()
}
```
